# Install
```bash
  python3 -m venv ./env
  pip3 install -r requirements.txt

  cd module/dupeguru
  python3 -m venv --system-site-packages ./env
  source ./env/bin/activate
  pip3 install -r requirements.txt
  python3 build.py
  chmod +x modules/dupeguru/run.py
```

# Run
```bash
  python3 MainWindow.py
```
